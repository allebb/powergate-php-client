170:     public function createRequest($method = 'GET', $uri = null, $headers = null, $body = null, array $options = array())
171:     {
172:         if (!$uri) {
173:             $url = $this->getBaseUrl();
174:         } else {
175:             if (!is_array($uri)) {
176:                 $templateVars = null;
177:             } else {
