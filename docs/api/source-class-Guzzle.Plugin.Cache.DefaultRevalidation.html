 31:     {
 32:         try {
 33:             $revalidate = $this->createRevalidationRequest($request, $response);
 34:             $validateResponse = $revalidate->send();
 35:             if ($validateResponse->getStatusCode() == 200) {
 36:                 return $this->handle200Response($request, $validateResponse);
 37:             } elseif ($validateResponse->getStatusCode() == 304) {
 38:                 return $this->handle304Response($request, $validateResponse, $response);
 39:             }
 40:         } catch (BadResponseException $e) {
 41:             $this->handleBadResponse($e);
 42:         }
 43: 
 44:         // Other exceptions encountered in the revalidation request are ignored
 45:         // in hopes that sending a request to the origin server will fix it
 46:         return false;
 47:     }

 49:     public function shouldRevalidate(RequestInterface $request, Response $response)
 50:     {
 51:         if ($request->getMethod() != RequestInterface::GET) {
 52:             return false;
 53:         }
 54: 
 55:         $reqCache = $request->getHeader('Cache-Control');
 56:         $resCache = $response->getHeader('Cache-Control');
 57: 
 58:         $revalidate = $request->getHeader('Pragma') == 'no-cache' ||
 59:             ($reqCache && ($reqCache->hasDirective('no-cache') || $reqCache->hasDirective('must-revalidate'))) ||
 60:             ($resCache && ($resCache->hasDirective('no-cache') || $resCache->hasDirective('must-revalidate')));
 61: 
 62:         // Use the strong ETag validator if available and the response contains no Cache-Control directive
 63:         if (!$revalidate && !$resCache && $response->hasHeader('ETag')) {
 64:             $revalidate = true;
 65:         }
 66: 
 67:         return $revalidate;
 68:     }

 88:      * Creates a request to use for revalidation
 89:      *
 90:      * @param RequestInterface $request  Request
 91:      * @param Response         $response Response to revalidate
 92:      *
 93:      * @return RequestInterface returns a revalidation request
 94:      */
 95:     protected function createRevalidationRequest(RequestInterface $request, Response $response)
 96:     {
 97:         $revalidate = clone $request;
 98:         $revalidate->removeHeader('Pragma')
 99:             ->removeHeader('Cache-Control')
100:             ->setHeader('If-Modified-Since', $response->getLastModified() ?: $response->getDate());
101: 
102:         if ($response->getEtag()) {
103:             $revalidate->setHeader('If-None-Match', $response->getEtag());
104:         }
105: 
106:         // Remove any cache plugins that might be on the request to prevent infinite recursive revalidations
107:         $dispatcher = $revalidate->getEventDispatcher();
108:         foreach ($dispatcher->getListeners() as $eventName => $listeners) {
109:             foreach ($listeners as $listener) {
110:                 if (is_array($listener) && $listener[0] instanceof CachePlugin) {
111:                     $dispatcher->removeListener($eventName, $listener);
112:                 }
113:             }
114:         }

117:     }

137:     /**
138:      * Handle a 304 response and ensure that it is still valid
139:      *
140:      * @param RequestInterface $request          Request that was sent
141:      * @param Response         $validateResponse Response received
142:      * @param Response         $response         Original cached response
143:      *
144:      * @return bool Returns true if valid, false if invalid
145:      */
146:     protected function handle304Response(RequestInterface $request, Response $validateResponse, Response $response)
147:     {
148:         static $replaceHeaders = array('Date', 'Expires', 'Cache-Control', 'ETag', 'Last-Modified');
149: 
150:         // Make sure that this response has the same ETag
151:         if ($validateResponse->getEtag() != $response->getEtag()) {
152:             return false;
153:         }
154: 
155:         // Replace cached headers with any of these headers from the
156:         // origin server that might be more up to date
157:         $modified = false;
158:         foreach ($replaceHeaders as $name) {
159:             if ($validateResponse->hasHeader($name)) {
160:                 $modified = true;
161:                 $response->setHeader($name, $validateResponse->getHeader($name));
162:             }
163:         }
164: 
165:         // Store the updated response in cache
166:         if ($modified && $this->canCache->canCacheResponse($response)) {
167:             $this->storage->cache($request, $response);
168:         }
169:
