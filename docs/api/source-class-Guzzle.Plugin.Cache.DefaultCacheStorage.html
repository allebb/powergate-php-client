 39:     public function cache(RequestInterface $request, Response $response)
 40:     {
 41:         $currentTime = time();
 42:         $ttl = $request->getParams()->get('cache.override_ttl') ?: $response->getMaxAge() ?: $this->defaultTtl;
 43: 
 44:         if ($cacheControl = $response->getHeader('Cache-Control')) {
 45:             $stale = $cacheControl->getDirective('stale-if-error');
 46:             $ttl += $stale == true ? $ttl : $stale;
 47:         }
 48: 
 49:         // Determine which manifest key should be used
 50:         $key = $this->getCacheKey($request);
 51:         $persistedRequest = $this->persistHeaders($request);
 52:         $entries = array();
 53: 
 54:         if ($manifest = $this->cache->fetch($key)) {
 55:             // Determine which cache entries should still be in the cache
 56:             $vary = $response->getVary();
 57:             foreach (unserialize($manifest) as $entry) {
 58:                 // Check if the entry is expired
 59:                 if ($entry[4] < $currentTime) {
 60:                     continue;
 61:                 }
 62:                 $entry[1]['vary'] = isset($entry[1]['vary']) ? $entry[1]['vary'] : '';
 63:                 if ($vary != $entry[1]['vary'] || !$this->requestsMatch($vary, $entry[0], $persistedRequest)) {
 64:                     $entries[] = $entry;
 65:                 }
 66:             }
 67:         }
 68: 
 69:         // Persist the response body if needed
 70:         $bodyDigest = null;
 71:         if ($response->getBody() && $response->getBody()->getContentLength() > 0) {
 72:             $bodyDigest = $this->getBodyKey($request->getUrl(), $response->getBody());
 73:             $this->cache->save($bodyDigest, (string) $response->getBody(), $ttl);
 74:         }
 75:

 77:             $persistedRequest,
 78:             $this->persistHeaders($response),
 79:             $response->getStatusCode(),
 80:             $bodyDigest,
 81:             $currentTime + $ttl
 82:         ));
 83: 
 84:         $this->cache->save($key, serialize($entries));
 85:     }

108:     public function fetch(RequestInterface $request)
109:     {
110:         $key = $this->getCacheKey($request);
111:         if (!($entries = $this->cache->fetch($key))) {
112:             return null;
113:         }
114: 
115:         $match = null;
116:         $headers = $this->persistHeaders($request);
117:         $entries = unserialize($entries);
118:         foreach ($entries as $index => $entry) {
119:             if ($this->requestsMatch(isset($entry[1]['vary']) ? $entry[1]['vary'] : '', $headers, $entry[0])) {
120:                 $match = $entry;
121:                 break;
122:             }
123:         }
124: 
125:         if (!$match) {
126:             return null;
127:         }
128: 
129:         // Ensure that the response is not expired
130:         $response = null;
131:         if ($match[4] < time()) {
132:             $response = -1;
133:         } else {
134:             $response = new Response($match[2], $match[1]);
135:             if ($match[3]) {
136:                 if ($body = $this->cache->fetch($match[3])) {
137:                     $response->setBody($body);
138:                 } else {
139:                     // The response is not valid because the body was somehow deleted
140:                     $response = -1;
141:                 }
142:             }
143:         }
144: 
145:         if ($response === -1) {
146:             // Remove the entry from the metadata and update the cache
147:             unset($entries[$index]);
148:             if ($entries) {
149:                 $this->cache->save($key, serialize($entries));
150:             } else {
151:                 $this->cache->delete($key);
152:             }
153:             return null;
154:         }

194:     /**
195:      * Determines whether two Request HTTP header sets are non-varying
196:      *
197:      * @param string $vary Response vary header
198:      * @param array  $r1   HTTP header array
199:      * @param array  $r2   HTTP header array
200:      *
201:      * @return bool
202:      */
203:     private function requestsMatch($vary, $r1, $r2)
204:     {
205:         if ($vary) {
206:             foreach (explode(',', $vary) as $header) {
207:                 $key = trim(strtolower($header));
208:                 $v1 = isset($r1[$key]) ? $r1[$key] : null;
209:                 $v2 = isset($r2[$key]) ? $r2[$key] : null;
210:                 if ($v1 !== $v2) {
211:                     return false;
212:                 }
213:             }
214:         }
215: 
216:         return true;
217:     }

229:         static $noCache = array(

231:             'connection' => true,
232:             'keep-alive' => true,

234:             'proxy-authorization' => true,

238:             'upgrade' => true,
239:             'set-cookie' => true,

243:         // Clone the response to not destroy any necessary headers when caching
244:         $headers = $message->getHeaders()->getAll();
245:         $headers = array_diff_key($headers, $noCache);
246:         // Cast the headers to a string
247:         $headers = array_map(function ($h) { return (string) $h; }, $headers);
248:
