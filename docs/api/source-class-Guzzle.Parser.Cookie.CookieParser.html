28:         $pieces = array_filter(array_map('trim', explode(';', $cookie)));
29: 
30:         // The name of the cookie (first kvp) must include an equal sign.
31:         if (empty($pieces) || !strpos($pieces[0], '=')) {
32:             return false;
33:         }

36:         $data = array_merge(array_fill_keys(array_keys(self::$cookieParts), null), array(
37:             'cookies'   => array(),
38:             'data'      => array(),
39:             'path'      => $path ?: '/',
40:             'http_only' => false,
41:             'discard'   => false,
42:             'domain'    => $host
43:         ));
44:         $foundNonCookies = 0;
45: 
46:         // Add the cookie pieces into the parsed data array
47:         foreach ($pieces as $part) {
48:

50:             $key = trim($cookieParts[0]);

53:                 // Can be a single value (e.g. secure, httpOnly)
54:                 $value = true;
55:             } else {

57:                 $value = trim($cookieParts[1], " \n\r\t\0\x0B\"");
58:                 if ($decode) {

60:                 }
61:             }
62:
